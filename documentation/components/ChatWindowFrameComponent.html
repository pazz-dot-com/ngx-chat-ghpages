import { Component, EventEmitter, Input, OnInit, Output } from '@angular/core';

@Component({
    selector: 'ngx-chat-window-frame',
    templateUrl: './chat-window-frame.component.html',
    styleUrls: ['./chat-window-frame.component.less'],
})
export class ChatWindowFrameComponent implements OnInit {

    @Output()
    fileDrop = new EventEmitter<File>();

    @Input()
    fileDropEnabled = false;

    @Input()
    fileDropMessageKey = '';

    @Input()
    windowIcon = '';

    @Input()
    windowTitle = '';

    @Output()
    closeClick = new EventEmitter<void>();

    @Output()
    headerClick = new EventEmitter<void>();

    constructor() { }

    ngOnInit(): void {
    }

}

<div class="chat-window">
    <div (click)="headerClick.emit()" class="chat-contact-header" [title]="windowTitle">
        <div class="chat-contact-avatar" *ngIf="windowIcon">
            <img [src]="windowIcon"/>
        </div>
        <div class="chat-contact-name">
            {{windowTitle}}
        </div>
        <div *ngIf="closeClick.observers.length > 0" class="chat-close" (click)="closeClick.emit()">
            &times;
        </div>
    </div>

    <ngx-chat-filedrop [enabled]="fileDropEnabled"
                       (fileUpload)="fileDrop.emit($event)"
                       [dropMessage]="fileDropMessageKey">
        <ng-content></ng-content>
    </ngx-chat-filedrop>
</div>
